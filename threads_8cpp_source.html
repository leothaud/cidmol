64struct [[gnu::aligned(16), gnu::packed]] stackHead {
66 void (*entry)(stackHead *);
68 TlsHeader *tlsHeader;
70 ExecutionFutex *futex;
72 void (*fn)(u8 *);
74 u8 *arg;
76 u8 *stack;
78 Logger *logger;
79};

82[[clang::noinline, clang::no_stack_protector]] void
83threadEntry(stackHead *head) {
84 // Initialize the thread logger.
85 getThreadLogger() = head->logger;
86 // Call the target function wrapper
87 head->fn(head->arg);
88 // Broadcast termination
89 if (head->futex) {
90 terminate(head->futex);
91 }
92
93 // Call the thread_local variable destructors.
94 auto &cxaAtExitWrapper = getCxaThreadAtExitWrapper();
95 for (u64 i = cxaAtExitWrapper.index; i > 0; --i) {
96 cxaAtExitWrapper.func[i](cxaAtExitWrapper.arg[i]);
97 cxaAtExitWrapper.func[i] = nullptr;
98 }
99
100 // Free the thread data and stack
101 // Warning: this will work with the current allocator
102 // and exit implementation. But may break if allocator change
103 delete[] head->tlsHeader->tlsInfo;
104 delete[] head->arg;
105 delete[] head->stack;
106 exit(0);
107}

111#if defined(linux) and defined(__x86_64__) and __has_extension(gnu_asm)
112 __asm__ volatile("mov %%rdi, %%rsi\n"
113 "mov 8(%%rsi), %%r8\n"
114 "mov $0xd0f00, %%edi\n"
115 "mov $56, %%eax\n"
116 "xor %%edx, %%edx\n"
117 "xor %%r10d, %%r10d\n"
118 "syscall\n"
119 "mov %%rsp, %%rdi\n"
120 "ret\n" ::
121 : "rax", "rcx", "rsi", "rdi", "r8", "r10", "r11",

124#error No alternative for newthread function.

133template <typename H, typename... T> struct ParameterPack<H, T...> {
134 H head;
135 ParameterPack<T...> tail;
136 ParameterPack(H head, T... tail) : head(head), tail(forward<T>(tail)...) {}
137};
138template <typename F, typename... ARGS> struct FunEntry {
139 F fun;
140 ParameterPack<ARGS...> args;
141
142 FunEntry(F fun, ARGS... args) : fun(fun), args(forward<ARGS>(args)...) {}
143};

145template <u64 I, typename T> struct ArgsBuilder;
146
147template <typename H, typename... T>
148struct ArgsBuilder<0, ParameterPack<H, T...>> {
149 static H &get(ParameterPack<H, T...> &a) { return a.head; }
150};
151
152template <u64 I, typename H, typename... T>
153struct ArgsBuilder<I, ParameterPack<H, T...>> {
154 static auto &get(ParameterPack<H, T...> &a) {
155 return ArgsBuilder<I - 1, ParameterPack<T...>>::get(a.tail);
156 }
157};
158
159template <typename F, typename... ARGS, u64... I>
160void doCallImpl(F &fun, ParameterPack<ARGS...> args, IndicesSequence<I...>) {
161 fun(ArgsBuilder<I, ParameterPack<ARGS...>>::get(args)...);
162}
164
166template <typename F, typename... ARGS>
167 requires(IsCallable<F, ARGS...>)
168void doCall(u8 *entry) {
169 auto *funEntry = reinterpret_cast<FunEntry<F, ARGS...> *>(entry);
170 doCallImpl(funEntry->fun, funEntry->args,
171 IndicesSequenceBuilder<sizeof...(ARGS)>{});
172 funEntry->~FunEntry();
173}

198export template <auto fn, typename... ARGS>
199 requires(IsCallable<decltype(fn), ARGS...>)
200[[clang::always_inline, clang::no_stack_protector]] void
201startThread(StartThreadOptions options, ARGS &&...args) {
202 static constexpr u64 stackSize = 8 * 1024 * 1024;
203 u8 *stack = new u8[stackSize];
204 stackHead *head = reinterpret_cast<stackHead *>(
205 reinterpret_cast<u64>((stack + stackSize - sizeof(stackHead))) & ~0xFllu);
206 u8 *entryPtr = new u8[sizeof(FunEntry<decltype(fn), ARGS...>)];
207 FunEntry<decltype(fn), ARGS...> *entry = new (entryPtr)
208 FunEntry<decltype(fn), ARGS...>(fn, forward<ARGS>(args)...);
209 head->logger = options.logger;
210 head->entry = threadEntry;
211 head->futex = options.futex;
212 head->fn = doCall<decltype(fn), ARGS...>;
213 head->arg = entryPtr;
214 head->stack = stack;
215 head->tlsHeader = allocateTls();
216
217 newthread(head);
218}

225startThread(ARGS &&...args) {
